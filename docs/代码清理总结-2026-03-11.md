#### 核心特性

```typescript
// ✅ 重构前
async function deleteJob(id: string) {
  const response = await fetch(`${API_BASE_URL}/api/jobs/${id}`, {
    method: 'DELETE',
  });
  if (!response.ok) {
    throw new Error('删除职位失败');
  }
  return { success: true };
}

// ✅ 重构后
async function deleteJob(id: string) {
  return del(`/api/jobs/${id}`);
}

// ✅ 在组件中使用
const { withErrorHandling } = useErrorHandler();

const handleDelete = async (id: string) => {
  await withErrorHandling(
    () => deleteJob(id),
    {
      showSuccessToast: true,
      successMessage: '删除成功',
    }
  );
};
```

#### 错误类型

| 错误类型 | HTTP 状态码 | 说明 |
|---------|------------|------|
| `NetworkError` | 0 | 网络连接失败 |
| `AuthError` | 401, 403 | 认证/授权失败 |
| `NotFoundError` | 404 | 资源不存在 |
| `BusinessError` | 400, 422 | 业务逻辑错误 |
| `TimeoutError` | 408 | 请求超时（30秒） |
| `ServerError` | 500, 502, 503 | 服务器错误 |
